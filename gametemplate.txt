  <script>
    function toggleMenu() {
        document.getElementById("indexMenu").classList.toggle("active");
    }

    const container = document.getElementById("flash-container");

    function resizeGame() {
      const windowWidth = window.innerWidth;
      const windowHeight = window.innerHeight;
      const aspectRatio = 4 / 3; // Standard Flash aspect ratio
      let width = windowWidth;
      let height = windowWidth / aspectRatio;
      if (height > windowHeight) {
        height = windowHeight;
        width = height * aspectRatio;
      }
      container.style.width = `${Math.floor(width)}px`;
      container.style.height = `${Math.floor(height)}px`;
    }

    window.addEventListener("resize", resizeGame);
    window.addEventListener("DOMContentLoaded", () => {
      resizeGame();
      const ruffle = window.RufflePlayer?.newest() || window.RufflePlayer?.createPlayer();
      if (ruffle && container) {
        const player = ruffle.createPlayer();
        player.style.width = "100%";
        player.style.height = "100%";
        container.appendChild(player);

        /* -----------------------------------------------------------
           CHANGE THE LINK BELOW TO THE NEW .SWF FILE YOU WANT TO LOAD
           ----------------------------------------------------------- */
        player.load("PASTE_YOUR_SWF_URL_HERE"); 

      } else {
        container.textContent = "Ruffle failed to load."; 
      }
    });
  </script>
